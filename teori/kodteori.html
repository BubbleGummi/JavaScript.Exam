            // Funktionen getGrade tar tre argument: studentPoints, minPointsForG och minPointsForVG
            function getGrade(studentPoints, minPointsForG, minPointsForVG) {
            // Om studentPoints är mindre än minPointsForG returneras "U"
            if (studentPoints < minPointsForG) { 
                return "U" ; 
            } 
            // Annars, om studentPoints är mindre än minPointsForVGreturneras "G" 
            else if (studentPoints < minPointsForVG) { 
                return "G" ; 
            } 
            // Annars returneras "VG" else {
                return "VG" ; 
            } 
            }
            // Anropar getGrade-funktionen med olika värden och visar resultatet i alert-rutor
            // Första anropet: "U" eftersom 5 < 11
            alert(getGrade(5, 11, 16));   
            // Andra anropet: "G" eftersom 11 < 16
            alert(getGrade(11, 11, 16));
            // Tredje anropet: "VG" eftersom 18 >= 16  
            alert(getGrade(18, 11, 16));  

            // Anropar getGrade-funktionen med värdena 8, 10, 16 och lagrar resultatet i variabeln grade
            var grade = getGrade(8, 10, 16);
            
            // Om grade är "U" visas en alert-ruta om att studenten behöver göra omexamination
            if (grade == "U") {
            alert("tyvärr du fick " + grade + " och behöver göra omexamination");
            }
            // Utskriften kommer visa U, G, VG, U och en alert-ruta om att studenten behöver göra omexamination
